    wait = function (target){
  if(!queue){
    throw new Error("Can not wait, the event-queue module is not available");
  }

      result = value;
    },

      isFinished = true;
      isError = true;
      result = error;

    if(isError){
      throw result;
    }
    return result;
  }

  }
